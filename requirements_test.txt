# OctoBox — dependencias de teste e qualidade.
#
# POR QUE EXISTE:
# - Centraliza pacotes usados em CI/local para rodar a suite de testes,
#   medir cobertura, gerar fakes/factories, mockar HTTP e relogio, e
#   detectar testes order-dependentes.
# - Mantem versoes pinadas (== em vez de >=) para que CI seja
#   reproduzivel — qualquer mudanca de pacote vira commit explicito.
#
# O QUE INCLUI:
# 1. pytest core (pytest, pytest-django, pytest-benchmark)
# 2. cobertura (pytest-cov, coverage)
# 3. mocks e parametrizacao auxiliares (pytest-mock, freezegun, responses)
# 4. test data factories (factory-boy traz Faker)
# 5. higiene da suite (pytest-randomly para detectar order-dependence)
#
# E2E browser tests (Fase 8 do plano de qualidade).
# pytest-playwright orquestra o browser; playwright instala os binarios
# separadamente via `playwright install --with-deps chromium` (so em CI).
# Testes marcados com @pytest.mark.e2e ficam em tests/e2e/ e sao ignorados
# pelo addopts padrao (--ignore=tests/e2e). O workflow e2e-nightly.yml
# roda explicitamente `pytest tests/e2e/` em push para main e nightly.
#
# PONTOS CRITICOS:
# - Versoes devem ser compativeis com Django==6.0.5 e Python 3.12.
# - Manter sincronizado com os pacotes que os workflows CI instalam.

# Core test runner.
pytest==9.0.3
pytest-django==4.12.0
pytest-benchmark==5.2.3

# Coverage measurement.
pytest-cov==7.1.0
coverage==7.14.0

# Mocking helpers.
pytest-mock==3.15.1
freezegun==1.5.5
responses==0.26.0

# Test data factories (factory-boy traz Faker como dependencia).
factory-boy==3.3.3

# Suite hygiene: detecta testes order-dependentes ao randomizar ordem.
pytest-randomly==4.1.0

# Paralelizacao de suite. Fase 1 do plano de qualidade: sem isso a suite
# leva ~125min, inviavel para CI. Com -n auto cai para 20-30min em 4-8
# cores. Trade-off: pytest-randomly fica desativado em paralelo (ordem
# nao e mais total por processo). Aceitavel: Fase 3 (order-dependence)
# valida com seed fixo em runs separados.
pytest-xdist==3.6.1

# E2E — Fase 8 do plano de qualidade de testes.
# 0.5.x–0.7.1 requerem pytest<9.0.0 (incompatível com pytest==9.0.3).
# 0.7.2 foi a primeira versão a aceitar pytest<10.0.0.
pytest-playwright==0.7.2

# O driver playwright e dependencia transitiva do pytest-playwright e nao
# vinha pinado — cada run de CI instalava a ultima versao, contrariando a
# filosofia deste arquivo (upgrade de pacote = commit explicito). Ao subir
# a versao, rodar tests/e2e/ local e o e2e-nightly.yml manual antes do merge.
playwright==1.60.0
